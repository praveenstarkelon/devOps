Namespaces list and creation

# kubectl get namespace या kubectl get ns
NAME                   STATUS   AGE
default                Active   18m
kube-node-lease        Active   18m
kube-public            Active   18m
kube-system            Active   18m
local-path-storage     Active   17m
यह सभी namespace दिखाता है।
# kubectl create ns nginx ― नया namespace "nginx" बनाएँ।
# kubectl del ns <name> ― किसी namespace को delete करें।
nginx नाम का namespace बन चुका है, अब इसमें चीज़ें बना सकते हैं।

Pods in Namespace

# kubectl get pod -n nginx ― सभी pods देखें इस namespace में।
# kubectl del pod nginx --n nginx ― pod delete करना सीखें।

Namespace YAML से बनाना

YAML file: nginx_namespace.yml

kind: Namespace
apiVersion: v1
metadata:
  name: nginx
# kubectl apply -f nginx_namespace.yml -- YAML से namespace बनाएं।
